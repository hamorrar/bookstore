# Register new user
# Success
curl -X POST \
-H "Content-Type: application/json" \
-d '{
  "email": "user1@gmail.com",
  "password": "password1",
  "role": "Customer"
}' \
-w "\nHTTP Status: %{http_code}\n" \
http://localhost:8080/api/v1/auth/register
